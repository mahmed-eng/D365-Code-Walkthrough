
Select top 10  TRANSFERID,INVENTLOCATIONIDTO,* from  INVENTTRANSFERTABLE where  INVENTLOCATIONIDTO = 'AKTPLZ' AND  TRANSFERID = 'TO-0000001'

Select ITT. TRANSFERID,ITT.INVENTLOCATIONIDTO,IL.InventLocationId,IL.NAME,* from INVENTTRANSFERTABLE  ITT
join INVENTLOCATION IL ON ITT.INVENTLOCATIONIDTO = IL.InventLocationId AND ITT.INVENTLOCATIONIDFROM = IL.InventLocationId AND  TRANSFERID = 'TO-0000001'
